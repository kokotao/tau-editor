# ARC-004: 文件同步和自动保存机制设计

## 1. 概述

本设计文档描述文本编辑器的文件同步和自动保存机制，确保用户数据的安全性和一致性。

### 核心目标

- **数据安全**: 防止意外丢失未保存的更改
- **性能优化**: 避免频繁写入影响性能
- **冲突处理**: 处理文件外部修改的情况
- **用户友好**: 提供清晰的状态反馈

### 2.2 自动保存流程

```typescript
// services/autoSave.ts
import { useEditorStore } from '@/stores/editor';
import { useTabsStore } from '@/stores/tabs';
import { useSettingsStore } from '@/stores/settings';
import { invoke } from '@tauri-apps/api/core';

class AutoSaveService {
  private timer: ReturnType<typeof setTimeout> | null = null;
  private debounceDelay = 1000; // 1 秒防抖
  private isSaving = false;

  constructor() {
    this.start();
  }

  // 启动自动保存
  start() {
    this.stop();
    const settingsStore = useSettingsStore();
    
    if (settingsStore.autoSaveEnabled && settingsStore.autoSaveInterval > 0) {
      this.scheduleNextSave();
    }
  }

  // 停止自动保存
  stop() {
    if (this.timer) {
      clearTimeout(this.timer);
      this.timer = null;
    }
  }

  // 调度下次保存
  private scheduleNextSave() {
    const settingsStore = useSettingsStore();
    const intervalMs = settingsStore.autoSaveInterval * 1000;

    this.timer = setTimeout(async () => {
      await this.saveAllDirty();
      this.scheduleNextSave();
    }, intervalMs);
  }

  // 保存所有脏标签
  private async saveAllDirty() {
    if (this.isSaving) return;

    const tabsStore = useTabsStore();
    const dirtyTabs = tabsStore.dirtyTabs;

    if (dirtyTabs.length === 0) return;

    this.isSaving = true;

    try {
      for (const tab of dirtyTabs) {
        await this.saveTab(tab.id);
      }
    } finally {
      this.isSaving = false;
    }
  }

  // 保存单个标签
  private async saveTab(tabId: string) {
    const tabsStore = useTabsStore();
    const editorStore = useEditorStore();

    // 如果当前标签不是激活的，需要临时切换
    const wasActive = tabsStore.activeTabId === tabId;
    if (!wasActive) {
      tabsStore.activateTab(tabId);
    }

    const content = editorStore.content;
    const tab = tabsStore.activeTab;

    if (!tab) return;

    try {
      await invoke('write_file', {
        path: tab.filePath,
        content,
      });

      tabsStore.updateTabDirty(tabId, false);
      editorStore.markAsSaved();

      // 发送保存成功通知
      this.notifySaveSuccess(tab.fileName);
    } catch (error) {
      this.notifySaveError(tab.fileName, error as string);
    }

    if (!wasActive) {
      tabsStore.activateTab(tabsStore.activeTabId!);
    }
  }

  // 内容变化时触发防抖保存
  onContentChange() {
    if (this.timer) {
      clearTimeout(this.timer);
    }

    this.timer = setTimeout(() => {
      this.saveCurrent();
    }, this.debounceDelay);
  }

  // 立即保存当前文件
  private async saveCurrent() {
    const tabsStore = useTabsStore();
    if (tabsStore.activeTabId) {
      await this.saveTab(tabsStore.activeTabId);
    }
  }

  // 通知保存成功
  private notifySaveSuccess(fileName: string) {
    // 可以通过状态栏或 toast 显示
    console.log(`[AutoSave] ${fileName} 已保存`);
  }

  // 通知保存失败
  private notifySaveError(fileName: string, error: string) {
    console.error(`[AutoSave] ${fileName} 保存失败：${error}`);
    // 显示错误提示
  }
}

export const autoSaveService = new AutoSaveService();
```

## 3. 文件变更检测

### 3.1 外部修改检测

当文件被外部程序修改时，需要检测并提示用户。

```typescript
// services/fileWatcher.ts
import { listen } from '@tauri-apps/api/event';

interface FileChange {
  path: string;
  type: 'modified' | 'deleted' | 'created';
  timestamp: number;
}

class FileWatcherService {
  private watchedFiles = new Map<string, {
    lastModified: number;
    content: string;
  }>();

  async start() {
    // 监听文件系统事件
    await listen<FileChange>('file-change', (event) => {
      this.handleFileChange(event.payload);
    });
  }

  // 添加监听文件
  watch(path: string, content: string) {
    this.watchedFiles.set(path, {
      lastModified: Date.now(),
      content,
    });
  }

  // 移除监听
  unwatch(path: string) {
    this.watchedFiles.delete(path);
  }

  // 处理文件变化
  private async handleFileChange(change: FileChange) {
    const watched = this.watchedFiles.get(change.path);
    if (!watched) return;

    switch (change.type) {
      case 'modified':
        await this.handleModified(change.path, watched);
        break;
      case 'deleted':
        await this.handleDeleted(change.path);
        break;
    }
  }

  // 处理文件修改
  private async handleModified(path: string, watched: { lastModified: number; content: string }) {
    const tabsStore = useTabsStore();
    const tabId = tabsStore.getTabIdByPath(path);
    
    if (!tabId) {
      // 文件已打开但不是当前标签
      return;
    }

    // 获取文件最新修改时间
    const fileInfo = await invoke<FileInfo>('get_file_info', { path });
    
    if (fileInfo.modified && fileInfo.modified > watched.lastModified) {
      // 文件被外部修改
      this.showFileModifiedDialog(path);
    }
  }

  // 显示文件修改对话框
  private showFileModifiedDialog(path: string) {
    // 显示对话框：重新加载 / 保留本地版本 / 忽略
    console.log(`文件 ${path} 被外部修改，请选择处理方式`);
  }

  // 处理文件删除
  private async handleDeleted(path: string) {
    const tabsStore = useTabsStore();
    const tabId = tabsStore.getTabIdByPath(path);

    if (tabId) {
      // 提示用户文件已被删除
      console.warn(`文件 ${path} 已被删除`);
    }
  }
}

export const fileWatcherService = new FileWatcherService();
```

### 3.2 Tauri 端文件监控

```rust
// src/file_watcher.rs
use notify::{Config, RecommendedWatcher, RecursiveMode, Watcher};
use std::sync::mpsc::channel;
use tauri::{AppHandle, Manager};

pub struct FileWatcher {
    watcher: RecommendedWatcher,
}

impl FileWatcher {
    pub fn new(app_handle: AppHandle) -> Self {
        let (tx, rx) = channel();

        let mut watcher = RecommendedWatcher::new(
            move |event| {
                tx.send(event).unwrap();
            },
            Config::default(),
        )
        .unwrap();

        // 启动事件处理线程
        std::thread::spawn(move || {
            while let Ok(event) = rx.recv() {
                for path in event.paths {
                    let change_type = match event.kind {
                        notify::EventKind::Modify(_) => "modified",
                        notify::EventKind::Remove(_) => "deleted",
                        notify::EventKind::Create(_) => "created",
                        _ => continue,
                    };

                    let _ = app_handle.emit("file-change", FileChangePayload {
                        path: path.to_string_lossy().to_string(),
                        change_type: change_type.to_string(),
                        timestamp: std::time::SystemTime::now()
                            .duration_since(std::time::UNIX_EPOCH)
                            .unwrap()
                            .as_millis() as u64,
                    });
                }
            }
        });

        FileWatcher { watcher }
    }

    pub fn watch(&mut self, path: &str) -> Result<(), String> {
        self.watcher
            .watch(std::path::Path::new(path), RecursiveMode::NonRecursive)
            .map_err(|e| e.to_string())
    }

    pub fn unwatch(&mut self, path: &str) -> Result<(), String> {
        self.watcher
            .unwatch(std::path::Path::new(path))
            .map_err(|e| e.to_string())
    }
}

#[derive(Clone, serde::Serialize)]
struct FileChangePayload {
    path: String,
    change_type: String,
    timestamp: u64,
}
```

### 4.1 防止写入损坏

```rust
// src/commands/file_ops.rs
use std::path::Path;
use tokio::fs;
use uuid::Uuid;

pub async fn write_file(path: String, content: String) -> Result<(), String> {
    let path = Path::new(&path);

    // 1. 验证路径
    validate_path(&path)?;

    // 2. 确保父目录存在
    if let Some(parent) = path.parent() {
        fs::create_dir_all(parent).await
            .map_err(|e| format!("创建目录失败：{}", e))?;
    }

    // 3. 创建临时文件 (同一目录下的原子操作)
    let temp_path = path.with_extension(format!(
        "{}.tmp",
        Uuid::new_v4().to_string()
    ));

    // 4. 写入临时文件
    fs::write(&temp_path, &content)
        .await
        .map_err(|e| format!("写入临时文件失败：{}", e))?;

    // 5. 创建备份 (如果原文件存在)
    if path.exists() {
        let backup_path = path.with_extension(format!(
            "{}.bak",
            Uuid::new_v4().to_string()
        ));
        fs::rename(path, &backup_path)
            .await
            .map_err(|e| format!("创建备份失败：{}", e))?;
    }

    // 6. 原子替换 (rename 是原子操作)
    fs::rename(&temp_path, path)
        .await
        .map_err(|e| {
            // 回滚备份
            if let Some(parent) = path.parent() {
                let _ = fs::rename(
                    parent.join(format!("{}.bak", path.file_name().unwrap().to_string_lossy())),
                    path
                );
            }
            format!("保存文件失败：{}", e)
        })?;

    // 7. 清理旧备份 (保留最近 5 个)
```

## 5. 会话恢复

### 5.1 保存会话状态

```typescript
// services/sessionManager.ts
import { useTabsStore } from '@/stores/tabs';
import { useSettingsStore } from '@/stores/settings';

interface SessionState {
  timestamp: number;
  tabs: Array<{
    filePath: string;
    language: string;
    cursorPosition: { line: number; column: number };
  }>;
  activeTabId: string | null;
  fileTreeRoot: string | null;
}

class SessionManager {
  private readonly STORAGE_KEY = 'editor-session';

  // 保存当前会话
  saveSession() {
    const tabsStore = useTabsStore();
    const settingsStore = useSettingsStore();

    const session: SessionState = {
      timestamp: Date.now(),
      tabs: tabsStore.tabs.map(tab => ({
        filePath: tab.filePath,
        language: tab.language,
        cursorPosition: { line: 1, column: 1 }, // 可从 editorStore 获取
      })),
      activeTabId: tabsStore.activeTabId,
      fileTreeRoot: settingsStore.restoreLastSession ? tabsStore.tabs[0]?.filePath : null,
    };

    localStorage.setItem(this.STORAGE_KEY, JSON.stringify(session));
  }

  // 恢复会话
  async restoreSession(): Promise<boolean> {
    const settingsStore = useSettingsStore();
    
    if (!settingsStore.restoreLastSession) {
      return false;
    }

    const sessionStr = localStorage.getItem(this.STORAGE_KEY);
    if (!sessionStr) {
      return false;
    }

    try {
      const session: SessionState = JSON.parse(sessionStr);
      const tabsStore = useTabsStore();

      // 恢复标签页
      for (const tabData of session.tabs) {
        try {
          const content = await invoke<string>('read_file', {
            path: tabData.filePath,
          });

          tabsStore.addTab({
            filePath: tabData.filePath,
            fileName: path.basename(tabData.filePath),
            language: tabData.language,
            isDirty: false,
          });
        } catch (e) {
          console.warn(`无法恢复文件：${tabData.filePath}`);
        }
      }

      // 恢复激活状态
      if (session.activeTabId && tabsStore.tabs.length > 0) {
        tabsStore.activateTab(tabsStore.tabs[0].id);
      }

      return true;
    } catch (e) {
      console.error('会话恢复失败:', e);
      return false;
    }
  }

  // 清除保存的会话
  clearSession() {
    localStorage.removeItem(this.STORAGE_KEY);
  }
}

export const sessionManager = new SessionManager();
```

```vue
import { computed } from 'vue';

const status = computed(() => {
  if (editorStore.isDirty) {
    return 'dirty';
  }
  return 'saved';

  'status-saved': status.value === 'saved',
  'status-dirty': status.value === 'dirty',
  'status-error': status.value === 'error',
}));

  // 计算下次自动保存时间

});
```

## 8. 错误处理

### 8.1 保存失败处理流程

```typescript
// 保存失败时的处理
async function handleSaveError(error: string) {
  const errorType = categorizeError(error);

  switch (errorType) {
    case 'PERMISSION_DENIED':
      showPermissionDialog();
      break;
    case 'DISK_FULL':
      showDiskFullWarning();
      break;
    case 'FILE_LOCKED':
      showFileLockedDialog();
      break;
    default:
      showGenericError(error);
  }

  // 保留未保存的内容
  // 提供"另存为"选项
}

function categorizeError(error: string): string {
  if (error.includes('权限') || error.includes('permission')) {
    return 'PERMISSION_DENIED';
  }
  if (error.includes('空间') || error.includes('disk')) {
    return 'DISK_FULL';
  }
  if (error.includes('锁定') || error.includes('locked')) {
    return 'FILE_LOCKED';
  }
  return 'UNKNOWN';
}
```
